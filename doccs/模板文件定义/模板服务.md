```csharp
    // 报告实例相关方法

    Task<Result<List<ReportInstance>>> GetReportInstancesAsync();

    ReportInstance CreateNewReportInstance(string templateId, string templateVersion);

    // 新增方法
    List<TemplateItem> GetTemplates();

```

## TemplateService 实现

`TemplateService` 是 `ITemplateService` 的具体实现，负责模板文件的读写和管理。

### 构造函数

```csharp
public TemplateService()
{
    // 获取输出目录
    _templatesDirectory = AppDomain.CurrentDomain.BaseDirectory;
    _baseTemplateDirectory = Path.Combine(_templatesDirectory, "baseTemplate");
    _actualTemplateDirectory = Path.Combine(_templatesDirectory, "Templates");
    _templatesIndexFile = Path.Combine(_templatesDirectory, "templates.json");
    _versionsDirectory = Path.Combine(_templatesDirectory, "Versions");
    _reportsDirectory = Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "Reports");

    EnsureDirectories();
    
    // 异步初始化当前模板，避免阻塞构造函数
    _ = InitializeCurrentTemplateAsync();
}

private void EnsureDirectories()
{
    if (!Directory.Exists(_templatesDirectory)) Directory.CreateDirectory(_templatesDirectory);
    if (!Directory.Exists(_baseTemplateDirectory)) Directory.CreateDirectory(_baseTemplateDirectory);
    if (!Directory.Exists(_actualTemplateDirectory)) Directory.CreateDirectory(_actualTemplateDirectory);
    if (!Directory.Exists(_versionsDirectory)) Directory.CreateDirectory(_versionsDirectory);
    if (!Directory.Exists(_reportsDirectory)) Directory.CreateDirectory(_reportsDirectory);
}
```

### 模板加载方法

```csharp
public async Task<Result<List<TemplateData>>> GetAllTemplatesAsync()
{
    try
    {
        var templates = new List<TemplateData>();
        
        // 扫描基础模板目录
        if (Directory.Exists(_baseTemplateDirectory))
        {
            var baseTemplateFiles = Directory.GetFiles(_baseTemplateDirectory, "*.json");
            foreach (var file in baseTemplateFiles)
            {
                var result = await GetTemplateByPathAsync(file);
                if (result.IsSuccess) templates.Add(result.Value);
            }
        }
        
        // 扫描检验项目模板目录
        if (Directory.Exists(_actualTemplateDirectory))
        {
            var actualTemplateFiles = Directory.GetFiles(_actualTemplateDirectory, "*.json", SearchOption.AllDirectories);
            foreach (var file in actualTemplateFiles)
            {
                var result = await GetTemplateByPathAsync(file);
                if (result.IsSuccess) templates.Add(result.Value);
            }
        }
        
        return Result<List<TemplateData>>.Success(templates);
    }
    catch (Exception ex)
    {
        return Result<List<TemplateData>>.Failure($"获取模板列表失败：{ex.Message}");
    }
}

private async Task<Result<TemplateData>> GetTemplateByPathAsync(string path)
{
    if (_templateCache.TryGetValue(path, out var cached)) return Result<TemplateData>.Success(cached);

    try
    {
        var json = await File.ReadAllTextAsync(path);
        var template = JsonSerializer.Deserialize<TemplateData>(json);
        if (template != null)
        {
            _templateCache[path] = template;
            return Result<TemplateData>.Success(template);
        }
        return Result<TemplateData>.Failure("解析失败");
    }
    catch (Exception ex)
    {
        return Result<TemplateData>.Failure(ex.Message);
    }
}

public async Task<Result<TemplateData>> LoadDefaultTemplateAsync()
{
    // 优先从基础模板目录加载 A4 竖向生化模板作为默认
    var a4Path = Path.Combine(_baseTemplateDirectory, "A4_Vertical_Biochemical.json");
    if (File.Exists(a4Path))
    {
        return await GetTemplateByPathAsync(a4Path);
    }

    return Task.FromResult(Result<TemplateData>.Success(CreateNewTemplate()));
}
```

### 模板保存方法

```csharp
public async Task<Result> SaveTemplateAsync(TemplateData template)
{
    try
    {
        template.ModifiedDate = DateTime.Now;
        var templateFile = Path.Combine(_templatesDirectory, $"{template.TemplateId}.json");
        var json = JsonSerializer.Serialize(template, new JsonSerializerOptions { WriteIndented = true });
        await File.WriteAllTextAsync(templateFile, json);
        
        // 更新缓存
        _templateCache[templateFile] = template;
        
        return Result.Success();
    }
    catch (Exception ex)
    {
        return Result.Failure($"保存模板失败：{ex.Message}");
    }
}
```

### 模板树结构方法

```csharp
public List<TemplateItem> GetTemplates()
{
    var root = new TemplateItem { Name = "模板库", IsDirectory = true, Children = new List<TemplateItem>() };
    
    // 添加基础模板目录
    var baseTemplateNode = new TemplateItem { Name = "基础模板", IsDirectory = true, Children = new List<TemplateItem>() };
    if (Directory.Exists(_baseTemplateDirectory))
    {
        foreach (var file in Directory.GetFiles(_baseTemplateDirectory, "*.json"))
        {
            baseTemplateNode.Children.Add(new TemplateItem {
                Name = Path.GetFileNameWithoutExtension(file),
                Path = file,
                IsDirectory = false
            });
        }
    }
    root.Children.Add(baseTemplateNode);
    
    // 添加检验项目模板目录
    var actualTemplateNode = new TemplateItem { Name = "检验项目模板", IsDirectory = true, Children = new List<TemplateItem>() };
    if (Directory.Exists(_actualTemplateDirectory))
    {
        foreach (var projectDir in Directory.GetDirectories(_actualTemplateDirectory))
        {
            var projectName = Path.GetFileName(projectDir);
            var projectNode = new TemplateItem { Name = projectName, IsDirectory = true, Children = new List<TemplateItem>() };
            
            foreach (var file in Directory.GetFiles(projectDir, "*.json"))
            {
                projectNode.Children.Add(new TemplateItem {
                    Name = Path.GetFileNameWithoutExtension(file),
                    Path = file,
                    IsDirectory = false
                });
            }
            
            actualTemplateNode.Children.Add(projectNode);
        }
    }
    root.Children.Add(actualTemplateNode);
    
    return new List<TemplateItem> { root };
}

public async Task<Result<List<TemplateData>>> GetTemplatesTreeAsync()
{
    try
    {
        var allTemplates = await GetAllTemplatesAsync();
        if (!allTemplates.IsSuccess) return allTemplates;

        var templates = allTemplates.Value;
        var rootTemplates = templates.Where(t => string.IsNullOrEmpty(t.ParentId)).ToList();
        var templateMap = templates.ToDictionary(t => t.TemplateId);

        // 构建树状结构
        foreach (var template in templates)
        {
            if (!string.IsNullOrEmpty(template.ParentId) && templateMap.TryGetValue(template.ParentId, out var parent))
            {
                if (!parent.ChildTemplateIds.Contains(template.TemplateId))
                {
                    parent.ChildTemplateIds.Add(template.TemplateId);
                }
            }
        }

        return Result<List<TemplateData>>.Success(rootTemplates);
    }
    catch (Exception ex)
    {
        return Result<List<TemplateData>>.Failure($"获取模板树失败：{ex.Message}");
    }
}
```
